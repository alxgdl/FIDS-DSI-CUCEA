    <script>

        // 1. TU ENLACE CSV DE GOOGLE SHEETS

        const csvUrl = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vTtJ6ZlMM6skUMzxHY2LYmtU3WBojxBxxP1k7_exlFwS4THkhsC9OAt6Iq4NjxoXDvnugZwRzTjA9ZI/pub?gid=1178925333&single=true&output=csv';


        // 2. CONFIGURACIÓN DE VISUALIZACIÓN

        const maxFilas = 15; // Exactamente las 15 que pediste


        async function cargarDatos() {

            try {

                // El timestamp '&t=' evita que el Fire Stick guarde caché viejo

                const respuesta = await fetch(csvUrl + '&t=' + Date.now());

                const texto = await respuesta.text();

                const filas = texto.split('\n');

                

                let tablaHTML = '<table><tbody>';


                // Empezamos en i=1 para saltar el encabezado de Google Sheets

                for (let i = 1; i <= maxFilas; i++) {

                    if (filas[i]) {

                        const columnas = filas[i].split(',');

                        tablaHTML += '<tr>';

                        columnas.forEach(col => {

                            tablaHTML += `<td>${col.trim()}</td>`;

                        });

                        tablaHTML += '</tr>';

                    } else {

                        // Si hay menos de 15 filas, crea filas vacías para mantener el diseño

                        tablaHTML += '<tr style="height:45px"><td></td><td></td><td></td><td></td><td></td></tr>';

                    }

                }


                tablaHTML += '</tbody></table>';

                document.getElementById('tabla-datos').innerHTML = tablaHTML;


            } catch (error) {

                console.error("Error al obtener CSV:", error);

            }

        }


        // Reloj en tiempo real

        function actualizarReloj() {

            const ahora = new Date();

            document.getElementById('reloj').innerText = ahora.toLocaleTimeString();

        }


        // Ejecución

        actualizarReloj();

        setInterval(actualizarReloj, 1000);

        

        cargarDatos();

        setInterval(cargarDatos, 60000); // Actualiza datos cada minuto
